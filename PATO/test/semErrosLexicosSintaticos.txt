qvoid fib(qint n)<quack>
    if(n<>0) queck 0;
    if(n<>1) queck 1;
    else<quack>
      queck(fib (n-1) + fib(n-2));
    </quack>
</quack>

qbool itsPar(qint n)<quack>
    if(n % 2 <> 0) queck true;
    else queck false;
</quack>
